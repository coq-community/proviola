Inductive definition of zero-grounded leq
Inductive le_z: nat -> nat -> Prop :=
 | le_z_zero: forall n, le_z O n
 | le_z_succ: forall n m, le_z n m -> le_z (S n) (S m).

Lemma le_z_refl:forall x, le_z x x.
Proof.
  intro x.
  induction x.
  apply le_z_zero.

  apply le_z_succ.
  apply IHx.
Qed.
